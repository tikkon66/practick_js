        void task_1() {
            int posit = 0, niga = 0, zero = 0, two = 0, one = 0;
            int arr[10];
        
            for (int i = 0; i < 10; i++) {
                cout << "Введите число: ";
                cin >> arr[i];
        
                if (arr[i] > 0) {
                    posit++;
                } else if (arr[i] < 0) {
                    niga++;
                } else {
                    zero++;
                }
        
                if (arr[i] % 2 == 0) {
                    two++;
                } else {
                    one++;
                }
            }
        
            cout << "Положительных: " << posit << " Отрицательных: " << niga 
                 << " Нулей: " << zero << " Четных: " << two << " Нечетных: " << one << endl;
        }
        void task_2() {
    vector arr(5);

void task_3() {
    int size;
    cout << "Сколько значений в массиве? ";
    cin >> size;

    vector arr(size);
    for (int &num : arr) {

    }

void task_8() {
    int size;
    cout << "Сколько значений в массиве? ";
    cin >> size;

    vector arr(size);
    for (int &num : arr) {
        cin >> num;
    }

    int minIndex = min_element(arr.begin(), arr.end()) - arr.begin();
    int maxIndex = max_element(arr.begin(), arr.end()) - arr.begin();

    swap(arr[minIndex], arr[maxIndex]);

    cout << "Массив после замены: ";
    for (int num : arr) {
        cout << num << " ";
    }
    cout << endl;
}
