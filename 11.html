void task_1() {
    int posit = 0, niga = 0, zero = 0, two = 0, one = 0;
    int arr[10];

    for (int i = 0; i < 10; i++) {
        cout << "Введите число: ";
        cin >> arr[i];

        if (arr[i] > 0) posit++;
        else if (arr[i] < 0) niga++;
        else zero++;

        if (arr[i] % 2 == 0) two++;
        else one++;
    }

    cout << "Положительных: " << posit << " Отрицательных: " << niga 
         << " Нулей: " << zero << " Четных: " << two << " Нечетных: " << one;
}

void task_3() {
    int size;
    cout << "Сколько значений в массиве? ";
    cin >> size;

    int* arr = new int[size];
    for (int i = 0; i < size; i++) cin >> arr[i];

    int temp = arr[0];
    arr[0] = arr[size - 1];
    arr[size - 1] = temp;

    cout << "Массив после замены: ";
    for (int i = 0; i < size; i++) cout << arr[i] << " ";

    delete[] arr;
}

void task_8() {
    int size;
    cout << "Сколько значений в массиве? ";
    cin >> size;

    int* arr = new int[size];
    for (int i = 0; i < size; i++) cin >> arr[i];

    int minIndex = 0, maxIndex = 0;
    for (int i = 1; i < size; i++) {
        if (arr[i] < arr[minIndex]) minIndex = i;
        if (arr[i] > arr[maxIndex]) maxIndex = i;
    }

    int temp = arr[minIndex];
    arr[minIndex] = arr[maxIndex];
    arr[maxIndex] = temp;

    cout << "Массив после замены: ";
    for (int i = 0; i < size; i++) cout << arr[i] << " ";

    delete[] arr;
}

    int index;

}

void task_10() {

void task_13() {
    int size;
    cout << "Сколько значений в массиве? ";
    cin >> size;

    char* arr = new char[size];
    for (int i = 0; i < size; i++) cin >> arr[i];

    char vowels[] = {'a', 'e', 'i', 'o', 'u'};
    int vowelCount = 0;

    for (int i = 0; i < size; i++) {
        for (char vowel : vowels) {
            if (arr[i] == vowel) {
                char temp = arr[i];
                arr[i] = arr[vowelCount];
                arr[vowelCount] = temp;
                vowelCount++;
                break;
            }
        }
    }

    cout << "Отсортированный массив: ";
    for (int i = 0; i < size; i++) cout << arr[i] << " ";

    delete[] arr;
}

void task_14() {
    int number;
    cout << "Введите число: ";
    cin >> number;

    int temp = number;
    int digitCount = 0;
    while (temp != 0) {
        temp /= 10;
        digitCount++;
    }

    int* arr = new int[digitCount];
    for (int i = digitCount - 1; i >= 0; i--) {
        arr[i] = number % 10;
        number /= 10;
    }

    cout << "Массив цифр: ";
    for (int i = 0; i < digitCount; i++) cout << arr[i] << " ";

    delete[] arr;
}
