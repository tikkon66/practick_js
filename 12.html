// 7) Создать двумерный массив NN-размерности, найти строки с максимальным и минимальным элементом и поменять их местами, если строки совпадают, обнулить их. Числа генерировать случайно.

void task_7() {
    int n, m, min, min_m = 0, min_n = 0, max, max_m = 0, max_n = 0;
    cout << "сколько столбов; ";
    cin >> m;
    cout << "сколько строк; ";
    cin >> n;
    cout << endl;
    
    int **arr = new int*[m];
    for (int i = 0; i < m; i++) {
        arr[i] = new int [n];
    }
            
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            arr[i][j] = rand() % 99;
        }
    }
    
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            cout << arr[i][j] << "\t";
        }
        cout << endl;
    }
    cout << endl;
    
    min = arr[0][0];
    max = arr[0][0];
    
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (min > arr[i][j]) {
                min = arr[i][j];
                min_m = i;
                min_n = j;
            }
            
            if (max < arr[i][j]) {
                max = arr[i][j];
                max_m = i;
                max_n = j;
            }
        }
    }
    
    if (max == min) {
        arr[min_m][min_n] = 0;
        arr[max_m][max_n] = 0;
    }
    else {
        arr[min_m][min_n] = max;
        arr[max_m][max_n] = min;
    }
    
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            cout << arr[i][j] << "\t";
        }
        cout << endl;
    }
}


// 8) Создать двумерный (пропорциональный) массив NN-размерности, Заполнить случайными числами и вывести его в виде матрицы. Затем отсортировать матрицу (верхний левый угол наименьшее число, нижний правый угол наибольшее число) и вывести измененную матрицу. 

void task_8() {
    int arr[7][7];
    
    for(int i = 0; i < 7; i++) {
        for (int j = 0; j < 7; j++) {
            arr[i][j] = rand() % 99;
        }
    }
    
    for (int i = 0; i < 7; i++) {
        for (int j = 0; j < 7; j++) {
            cout << arr[i][j] << "\t";
        }
        cout << endl;
    }
        cout << endl;
    
    int m = 0, n = 0, ok = true, min_m = 0, min_n = 0, min;
    
    for (int k = 0; k < 49; k++) {
        for (int i = m; i < 7; i++) {
            for (int j = n; j < 7; j++) {
                if (arr[min_m][min_n] > arr[i][j]) {
                    min_m = i;
                    min_n = j;
                }
            }
        }
        
        min = arr[min_m][min_n];
        arr[min_m][min_n] = arr[m][n];
        arr[m][n] = min;
        
        if (m == 0 && ok) {
            n++;
            ok = false;
        }
        else if (ok) {
            m--;
            n++;
        }
        else if (n == 0 && !ok) {
            m++;
            ok = true;
        }
        else if (!ok) {
            m++;
            n--;
        }
    }
    
    for (int i = 0; i < 7; i++) {
        for (int j = 0; j < 7; j++) {
            cout << arr[i][j] << "\t";
        }
        cout << endl;
    }
    
}
